def my_kernel_overloading():

    my_kernel_ = my_kernel.create()
    print(my_kernel_.k(x[0], y[0]))

    my_kernel.set_kernel_ptr(my_kernel_)
    my_kernel_ptr = core.kernel_interface.get_kernel_ptr()
